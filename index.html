import { db, auth } from "./firebase.js"; import { collection, addDoc, getDocs } from "https://www.gstatic.com/firebasejs/10.13.2/firebase-firestore.js"; window.createPost = async function () { const content = document.getElementById("postContent").value; if (!content) { alert("Nhập nội dung"); return; } if (!auth.currentUser) { alert("Hãy đăng nhập"); return; } await addDoc( collection(db, "posts"), { content: content, nickname: auth.currentUser.email.split("@")[0], email: auth.currentUser.email, userId: auth.currentUser.uid, likes: 0, createdAt: new Date().toISOString() } ); alert("Đăng bài thành công"); document.getElementById( "postContent" ).value = ""; loadPosts(); }; async function loadPosts() { const snapshot = await getDocs( collection(db, "posts") ); let html = ""; snapshot.forEach(doc => { const post = doc.data(); html += `

🌌 ${post.nickname}

${post.content}

❤️ ${post.likes}

`; }); document.getElementById( "posts" ).innerHTML = html; } window.addEventListener( "load", loadPosts );
